let transform = (n1, n2, res_len) => {
    n1.push(...new Array(res_len).fill(0));
    n1 = n1.slice(0, res_len);
    
    n2.push(...new Array(res_len).fill(0));
    n2 = n2.slice(0, res_len);
    return [n1, n2];
};

let sum = (n1, n2) => {
    res_len = Math.max(n1.length, n2.length) + 1;
    [n1, n2] = transform(n1, n2, res_len)
    
    let res = [];
    let to_mind = 0;
    for (let i = 0; i < res_len; i++) {
        let tmp = n1[i] + n2[i] + to_mind;
        res.push(tmp % 10);
        to_mind = Math.floor(tmp / 10);
    };
    return res.reverse().join('').replace(/^0+/, '');
};

    for(p = end; p > num; p--) {
        dig = *(p - 1) - '0';
        dig += add;
        *p  = (char)(dig % 10) + '0';
        add = (char)(dig / 10);

    {
    for (auto digit:bigint.number){
    stream<<digit;
    }
    return stream;
    }

BigInt operator+(const BigInt& lh, const BigInt& rh) {
    string str;
    int over=0,l,r;
int lim=max(lh.size(),rh.size());

    for(int i=0;i<lim;i++ ){ //for идет до максимальной длины числа
    if(i+1>lh.size())
            l=0;             // Если число короче второго - дополняем нулями
    else l=lh.number[lh.size()-i-1]-'0';

    if(i+1>rh.size())
            r=0;
    else r=rh.number[rh.size()-i-1]-'0';
    int tmp=l+r+over;
    str=(char)(tmp%10+'0')+str;
    over=tmp/10;
    }
if(over){
    str="1"+str;
}
return BigInt(str);
}
