Удали пробелы

Дана строка, Вам требуется преобразовать все идущие подряд пробелы в один. Если пробелы есть в начале и в конце строки - их нужно убрать.

Входные данные
Длина строки не превосходит 1000.

Выходные данные
Выведите измененную строку.

Sample Input:
nz d urp lren s bwz  boom  t a   j    ho    vi

Sample Output:
nz d urp lren s bwz boom t a j ho vi

Напишите программу. Тестируется через stdin → stdout

# C#
using System;
using System.Text;
using System.Linq;

public class MainClass
{
    public static void Main()
    {
        string s=Console.ReadLine();
        string k=String.Join(" ", s.Split(new string[] { " " }, StringSplitOptions.RemoveEmptyEntries));
        char [] chars = k.ToCharArray();
        StringBuilder sb=new StringBuilder();
        for (int i=0;i<chars.Length;i++)
        {
            if(chars[0]!=' '||chars[chars.Length-1]!=' ')
            {
                sb.Append(chars[i]);
            }
        }
        Console.WriteLine(sb);
    }
}

# C 
#include <stdbool.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define BUFF_SIZE 10000

char* removeExtraSpaces(char* const str) {
  if (!str) return NULL;
  size_t len = strlen(str);
  bool spaceFound = false;
  for (size_t i = len - 1; str[i] == ' ' || str[i] == '\n'; --i) {
    str[i] = '\0';
    --len;
  }
  char* edit = str;
  char* read = str;
  while (*read == ' ') {
    read++;
  }
  while (*read) {
    if (*read == ' ') {
      if (spaceFound) {
        read++;
        len--;
      } else {
        spaceFound = true;
        if (edit != read) *edit = *read;
        edit++;
        read++;
      }
    } else {
      if (spaceFound) spaceFound = false;
      if (edit != read) *edit = *read;
      edit++;
      read++;
    }
  }
  *edit = '\0';
}

int main(void) {
  char str[BUFF_SIZE];
  if (!fgets(str, BUFF_SIZE, stdin)) return EXIT_FAILURE;
  if (!removeExtraSpaces(str)) return EXIT_FAILURE;
  printf("%s\n", str);
  return EXIT_SUCCESS;
}
