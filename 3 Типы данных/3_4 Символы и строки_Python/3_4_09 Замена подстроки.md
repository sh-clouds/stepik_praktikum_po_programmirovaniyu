# C 
#include <stdio.h>

int main() {
    char c;
    while (scanf("%c", &c) == 1) {
        if (c == '1') {
            printf("one");
        } else {
            printf("%c", c);
        }
    }
    return 0;
}


# C++
#include <iostream>

int main() {
    std::string s, res = "";
    std::getline(std::cin, s);
    for (int i = 0; i < s.size(); i++) {
        if (s[i] == '1') res += "one";
        else res += s[i];
    }
    std::cout << res << "\n";
    return 0;
}
