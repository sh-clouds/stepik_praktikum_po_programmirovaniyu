# C#
using System;

public class MainClass
{
    public static void Main()
    {
        string line = Console.ReadLine();
        int index = 0;

        while (line.IndexOf('@') != -1)
        {
            index = line.IndexOf('@');
            line = line.Remove(index, 1);
        }
        Console.WriteLine(line);
    }
}
