# C#
using System;

public class LetterReplacing
{
    public static void Main()
    {
        char[] chars = Console.ReadLine().ToCharArray();
        
        bool hasFirstChar = false;
        int lastIndex = default;
        for (int i = 0; i < chars.Length; i++)
        {
            if (chars[i] != 'h') continue;
            if (!hasFirstChar) { hasFirstChar = true; continue; }
            
            chars[i] = 'H';
            lastIndex = i;
        }
        chars[lastIndex] = 'h';
        
        Console.WriteLine(chars);
    }
}


# java
import java.util.Scanner;

class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        String data = input.nextLine();

        int start = data.indexOf("h");
        int end = data.lastIndexOf("h");

        String theFirst = data.substring(0, start + 1);
        String theSecond = data.substring(start + 1, end).replace("h", "H");
        String theThird = data.substring(end);

        System.out.printf("%s%s%s", theFirst, theSecond, theThird);
    }
}

#include <iostream>

int main() {
    std::string s;
    std::getline(std::cin, s);
    
    int pos;
    while ((pos = s.find("h")) != std::string::npos) {
        s.replace(pos, 1, "H");
    }
    if (s.find("H") != -1) {
        s[s.find("H")] = 'h';
        s[s.rfind("H")] = 'h';
    }
    
    std::cout << s << "\n";
    return 0;
}
