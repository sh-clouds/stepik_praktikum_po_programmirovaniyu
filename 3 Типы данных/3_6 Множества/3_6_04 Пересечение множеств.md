Пересечение множеств

Даны два списка чисел, которые могут содержать до 10000 чисел каждый. Выведите все числа, которые входят как в первый, так и во второй список в порядке возрастания.

Примечание. И даже эту задачу на Питоне можно решить в одну строчку.

Входные данные
Вводятся два списка целых чисел. Все числа каждого списка находятся на отдельной строке.

Выходные данные
Выведите ответ на задачу.

Sample Input:
1 3 2
4 3 2

Sample Output:
2 3

# C#
using System;
using System.Collections.Generic;
using System.Linq;

public static class Program
{
    public static void Main()
    {
        var stroka1 = Console.ReadLine().Split(' ').Select(a => int.Parse(a)).ToArray();
        var stroka2 = Console.ReadLine().Split(' ').Select(a => int.Parse(a)).ToArray();
        
        var result = stroka1.Intersect(stroka2).OrderBy(a => a);
        foreach(var i in result)
            Console.Write(i + " ");
    }
}

#include <iostream>
#include <string>
#include <set>
#include <algorithm>
#include <sstream>

int main() {
    // put your code here
    std::string strFir;
    std::string strSec;
    
    //Считываем строки
    std::getline(std::cin, strFir);
    std::getline(std::cin, strSec);
    
    std::stringstream issFir (strFir);
    std::stringstream issSec (strSec);
   
    //Переводим в set
    int number;
    std::set <int> setFir;
    std::set <int> setSec;
    while(issFir >> number)
        setFir.insert(number);
        
    while(issSec >> number)
        setSec.insert(number);
    
    std::set <int> insertV;
    
    std::set_intersection(setFir.begin(), setFir.end(), setSec.begin(), setSec.end(), std::inserter(insertV, insertV.begin()));
    
    for (auto it : insertV)
       std::cout << it << " ";
    return 0;
}
