Сумма цифр

Дано трехзначное число (оно может быть отрицательным). Найдите сумму его цифр.

Входные данные
Вводится трехзначное число.

Выходные данные
Выведите ответ на задачу.

Sample Input:
192

Sample Output:
12

# C
#include <stdio.h>
#include <math.h>

int main() {
    int n, sum = 0; scanf("%d", &n);
    
    for(n = abs(n); n != 0; n /= 10) {
        sum += n % 10;
    }
    return printf("%d", sum) && 0;
}


# C++
#include <iostream>

int main() {
    int n, sum = 0;
    std::cin >> n;
    n = abs(n);
    while (n) {
        sum += n % 10;
        n /= 10;
    }
    std::cout << sum << "\n";
    return 0;
}
