# C
#include <stdio.h>
#define MAX(X, Y) (((X) > (Y)) ? (X) : (Y))

int main() {
    int a, b, c;
    scanf("%d %d %d", &a, &b, &c);
    
    int max = MAX(a, MAX(b, c));
    
    return printf("%d", max) && 0;
}
