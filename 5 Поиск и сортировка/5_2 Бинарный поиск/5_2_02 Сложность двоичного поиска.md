#  ruby
def findCnt(num_list, number, first, last, cnt)
    cnt += 1
    if last >= first
        middle = first + (last - first) / 2
        if num_list[middle] == number
            return cnt
        elsif num_list[middle] < number
            return findCnt(num_list, number, middle + 1, last, cnt)
        else
            return findCnt(num_list, number, first, middle - 1, cnt)    
        end
    else
        return -1
    end
end

# java
import java.util.Scanner;
import java.util.stream.IntStream;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int number = Integer.parseInt(input.nextLine());
        int[] array = IntStream.rangeClosed(1, number).toArray();
        System.out.println(search(array));
    }

    public static int search(int[] array) {
        int counter = 0, start = 0, middlePoint;

        while (start <= array.length - 1) {
            ++counter;
            middlePoint = (start + array.length - 1) / 2;
            start = middlePoint + 1;
        }

        return counter;
    }
}
