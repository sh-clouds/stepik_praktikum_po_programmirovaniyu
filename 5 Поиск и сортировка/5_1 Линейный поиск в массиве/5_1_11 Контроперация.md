Контроперация

Хакер Василий получил доступ к классному журналу и хочет заменить все свои минимальные оценки на максимальные. Напишите программу, которая заменяет оценки Василия, но наоборот (все максимальные - на минимальные).

Входные данные
Дано количество оценок Василия (не больше 100), затем сами оценки.

Выходные данные
Требуется вывести исправленные оценки в том же порядке.

Sample Input:
5 1 3 3 3 4

Sample Output:
1 3 3 3 1

Напишите программу. Тестируется через stdin → stdout

using System;
using System.Collections.Generic;
using System.Linq;

public class MainClass
{
    public static void Main()
    {
        var arr = new List<int>(Array.ConvertAll(Console.ReadLine().Split(), int.Parse));
        arr.RemoveAt(0);
        int max=arr.Max();
        int min=arr.Min();

        for (int i = 0; i < arr.Count; i++)
        {
            if (arr[i] == max)
                Console.Write(min + " ");
            else
                Console.Write(arr[i] + " ");
        }
    }
}


# java
import java.util.Arrays;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;
import java.util.stream.Collectors;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        input.nextInt();
        
        List<Integer> list = Arrays
                .stream(input.nextLine().trim().split("\\s+"))
                .map(Integer::parseInt)
                .collect(Collectors.toList());

        Integer min = Collections.min(list);
        Integer max = Collections.max(list);

        list.replaceAll(x -> x.equals(max) ? min : x);
        list.forEach(x -> System.out.print(x + " "));
    }
}

void change_array(int64_t** arr, size_t N);

int main() {
    size_t N = 0;

        change_array(&array, N);

}

void change_array(int64_t** arr, size_t N) {
    int64_t max = get_max(&(*arr), N);
    int64_t min = get_min(&(*arr), N);
    for (size_t i = 0; i < N; i++)
        if ((*arr)[i] == max) (*arr)[i] = min;
}

int64_t get_min(int64_t** arr, size_t N) {
    int64_t min = 5;
    for (size_t i = 0; i < N; i++)
        if (min > (*arr)[i]) min = (*arr)[i];
    return min;
}

int64_t get_max(int64_t** arr, size_t N) {
    int64_t max = 0;
    for (size_t i = 0; i < N; i++)
        if (max < (*arr)[i]) max = (*arr)[i];
    return max;
}

void echo_array(int64_t** arr, size_t N) {
    for (size_t i = 0; i < N; i++)
        printf("%"PRId64"%s", (*arr)[i], i + 1 == N ? "" : " ");
}

void input(size_t* N, int64_t** arr) {

    if ((*N))
        for (size_t i = 0; i < (*N); i++)
            scanf("%"SCNd64, &((*arr)[i]));
}
