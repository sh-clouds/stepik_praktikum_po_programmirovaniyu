Линейный поиск - 3

Напишите программу, которая выводит номера элементов массива, равных данному числу.

Входные данные
В первой строке задается одно натуральное число N, не превосходящее 1000 – размер массива.
Во второй строке вводятся N чисел – элементы массива (целые числа, не превосходящие по модулю 1000).
В третьей строке содержится одно целое число x, не превосходящее по модулю 1000.

Выходные данные
Вывести через пробел номера элементов, равных данному, в порядке возрастания. Если таких элементов нет, ничего выводить не нужно.

Sample Input:
5
1 2 3 4 5
3

Sample Output:
3

Напишите программу. Тестируется через stdin → stdout

# java
import java.util.Arrays;
import java.util.Scanner;
import java.util.stream.IntStream;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        int length = Integer.parseInt(input.nextLine().trim());
        int[] array = Arrays.stream(input.nextLine().split("\\s+")).mapToInt(Integer::parseInt).toArray();
        int number = Integer.parseInt(input.nextLine().trim());

        IntStream.iterate(0, x -> x + 1)
                .limit(length)
                .filter(x -> array[x] == number)
                .map(x -> x + 1)
                .forEach(x -> System.out.print(x + " "));
    }
}

int64_t nearest_x(int64_t** arr, size_t N, int64_t X) {
    int64_t item, delta = LLONG_MAX;
    for (size_t i = 0; i < N; i++)
        if (abs(X - (*arr)[i]) < delta) {
            item = (*arr)[i];
            delta = abs(X - (*arr)[i]);
        }
    return item;
}

size_t count_x(int64_t** arr, size_t N, int64_t X) {
    size_t count = 0;
    for (size_t i = 0; i < N; i++)
        if (X == (*arr)[i]) count++;
    return count;
}

void echo_array(int64_t** arr, size_t N) {
    for (size_t i = 0; i < N; i++)
        printf("%"PRId64"%s", (*arr)[i], i + 1 == N ? "" : " ");
}

    if (*N > 0)
        for (size_t i = 0; i < *N; i++)
            scanf("%"SCNd64, &((*arr)[i]));

}
