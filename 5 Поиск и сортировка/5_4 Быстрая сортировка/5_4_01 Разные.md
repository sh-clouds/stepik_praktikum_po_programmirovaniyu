Разные

Дано N чисел, требуется выяснить, сколько среди них различных.

Входные данные
В первой строке дано число N – количество чисел. (1 <= N <= 100000) Во второй строке даны через пробел N чисел, каждое не превышает 2*109 по модулю.

Выходные данные
Выведите число, равное количеству различных чисел среди данных.

Sample Input:
5
1 0 1 2 0

Sample Output:
3

def quick_sort(lst):
    if len(lst) < 2:
        return lst
    mid = lst[len(lst) // 2]
    center = [n for n in lst if n == mid]
    left = list(filter(lambda n: n < mid, lst))
    right = list(filter(lambda n: n > mid, lst))
    
    return quick_sort(left) + center + quick_sort(right)

#include <iostream>
#include <vector>
using namespace std;

void QuickSorting(vector<int>& arr, int start, int end) {
    int i = start;
    int pivot = end;
    while (i < pivot) {
        if (arr[i] > arr[pivot] && i == pivot - 1) {
            swap(arr[i], arr[pivot]);
            pivot--;
        }
        else if (arr[i] > arr[pivot]) {
            swap(arr[pivot], arr[pivot - 1]);
            swap(arr[i], arr[pivot]);
            pivot--;
        }
        else i++;
    }
    if (start < end) {
        QuickSorting(arr, start, pivot - 1);
        QuickSorting(arr, pivot + 1, end);
    }
}

int main() {
    int N; cin >> N;
    vector<int> arr(N);
    for (int i = 0; i < N; i++) cin >> arr[i];
    QuickSorting(arr, 0, N - 1);

    int unique = 1;
    for (int i = 1; i < N; i++) {
        if (arr[i] != arr[i - 1]) unique++;
    }
    cout << unique <<  "\n\n";
    return 0;
}
