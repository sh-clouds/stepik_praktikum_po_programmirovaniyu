Сортировка выбором максимума

Требуется отсортировать массив по неубыванию методом "выбор максимума".

Входные данные
В первой строке вводится одно натуральное число N, не превосходящее 1000 – размер массива. Во второй строке задаются N чисел – элементы массива (целые числа, не превосходящие по модулю 1000).

Выходные данные
Вывести получившийся массив.

Sample Input:
2
3 1

Sample Output:
1 3

Напишите программу. Тестируется через stdin → stdout

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int number = Integer.parseInt(input.nextLine());

        int[] array = new int[number];

        for (int a = 0; a < number; a++) {
            array[a] = input.nextInt();
        }

        array = selectionSort(array);

        for (int b = 0; b < number; b++) {
            if(b < number - 1) System.out.print(array[b] + " ");
            else System.out.println(array[b]);
        }
    }

    private static int[] selectionSort(int[] array) {
        for(int a = 0; a < array.length - 1; a++) {
            int index = a;

            for(int b = a + 1; b < array.length; b++) {
                if(array[b] < array[index]) {
                    index = b;
                }
            }

            int smallerNumber = array[index];
            array[index] = array[a];
            array[a] = smallerNumber;
        }
        return array;
    }
}

void echo_arr(int16_t** arr, size_t n) {
    for (int i = 0; i < n; i++) {
        printf("%"PRId16, (*arr)[i]);
        if (i + 1 != n) printf(" ");
    }
}

size_t find_imax(int16_t** arr, size_t n) {
    size_t imax = 0;
    int16_t max = SHRT_MIN;
    for (int i = 0; i < n; i++) {
        if ((*arr)[i] > max) {
            imax = i;
            max = (*arr)[i];
        }
    }
    return imax;
}

void sort_sel_max(int16_t** arr, size_t n) {
    while (n > 1) {
        size_t imax = find_imax(&(*arr), n);
        if (imax != n - 1) {
            int16_t swap = (*arr)[imax];
            (*arr)[imax] = (*arr)[n-1];
            (*arr)[n-1] = swap;
        }
        n--;
    }
}

int8_t init_arr(int16_t** arr, size_t n) {
    int8_t res = 0;
    *arr = malloc(n * sizeof(int8_t));
    if (*arr != NULL) {
        res = 1;
        for (int i = 0; i < n; i++)
            scanf("%"SCNd16, &((*arr)[i]));
    }
    return res;
}

size_t input_n() {
    size_t n = 0;
    scanf("%zu", &n);
    return n;
}
