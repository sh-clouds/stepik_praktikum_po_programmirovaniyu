Сортировка вставками

Требуется отсортировать массив по неубыванию методом "вставок".

Входные данные
В первой строке вводится одно натуральное число, не превосходящее 1000 – размер массива. Во второй строке задаются N чисел – элементы массива (целые числа, не превосходящие по модулю 1000).

Выходные данные
Вывести получившийся массив.

Sample Input:
5
5 4 3 2 1

Sample Output:
1 2 3 4 5

Напишите программу. Тестируется через stdin → stdout

# java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int number = Integer.parseInt(input.nextLine());

        int[] array = new int[number];

        for (int a = 0; a < number; a++) {
            array[a] = input.nextInt();
        }

        array = doInsertionSort(array);

        for (int b = 0; b < number; b++) {
            if(b < number - 1) System.out.print(array[b] + " ");
            else System.out.println(array[b]);
        }
    }

    public static int[] doInsertionSort(int[] array) {
        for(int a = 1; a < array.length; a++) {
            int key = array[a];
            int position = a - 1;

            while(position >= 0 && array[position] > key) {
                array[position + 1] = array[position];
                position--;
            }

            array[position + 1] = key;
        }
        return array;
    }
}

void sort_insertion(int16_t** arr, size_t n) {
    for (size_t i = 1; i < n; i++) {
        size_t j = i;
        while(j > 0 && (*arr)[j - 1] > (*arr)[j]) {
            int16_t swap = (*arr)[j - 1];
            (*arr)[j - 1] = (*arr)[j];
            (*arr)[j] = swap;
            j--;
        }
    }
}

void echo_arr(int16_t** arr, size_t n) {
    for (size_t i = 0; i < n; i++)
        printf("%"PRId16"%s", (*arr)[i], (i + 1 != n) ? " " : "");
}

int8_t init_arr(int16_t** arr, size_t n) {
    int8_t res = 0;
    *arr = malloc(n * sizeof(int16_t));
    if (*arr != NULL) {
        res = 1;
        for (size_t i = 0; i < n; i++)
            scanf("%"SCNd16, &((*arr)[i]));
    }
    return res;
}

size_t input_n() {
    size_t n = 0;
    scanf("%zu", &n);
    return n;
}
