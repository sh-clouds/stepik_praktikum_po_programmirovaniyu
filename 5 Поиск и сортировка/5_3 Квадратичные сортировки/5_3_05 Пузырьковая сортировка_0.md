Входные данные

Выходные данные
Вывести получившийся массив.

Sample Input:
5
5 4 3 2 1

Sample Output:
1 2 3 4 5

# java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int number = Integer.parseInt(input.nextLine());

        int[] array = new int[number];

        for (int a = 0; a < number; a++) {
            array[a] = input.nextInt();
        }

        array = doBubbleSort(array);

        for (int b = 0; b < number; b++) {
            if(b < number - 1) System.out.print(array[b] + " ");
            else System.out.println(array[b]);
        }
    }

    public static int[] doBubbleSort(int[] array) {
        for (int e = 0; e < array.length - 1; e++) {
            for (int f = 0; f < array.length - 1 - e; f++) {
                if (array[f] > array[f + 1]) {
                    int temporary = array[f];
                    array[f] = array[f + 1];
                    array[f + 1] = temporary;
                }
            }
        }
        return array;
    }
}

#include <inttypes.h>
#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

size_t input_n();
int8_t init_arr(int16_t** arr, size_t n);
void echo_arr(int16_t** arr, size_t n);
void sort_bubble(int16_t** arr, size_t n);

int main() {
    size_t n = input_n();
    int16_t* array = NULL;
    if (n > 0 && init_arr(&array, n)) {
        sort_bubble(&array, n);
        echo_arr(&array, n);
        free(array);
    }
    return 0;
}

void sort_bubble(int16_t** arr, size_t n) {
    for (size_t i = 0; i < n - 1; i++) {
        for (size_t j = 0; j < n - 1 - i; j++) {
            if ((*arr)[j] > (*arr)[j + 1]) {
                int16_t swap = (*arr)[j];
                (*arr)[j] = (*arr)[j + 1];
                (*arr)[j + 1] = swap;
            }
        }
    }
}

void echo_arr(int16_t** arr, size_t n) {
    for (size_t i = 0; i < n; i++)
        printf("%"PRId16"%s", (*arr)[i], (i + 1 != n) ? " " : "");
}

int8_t init_arr(int16_t** arr, size_t n) {
    int8_t res = 0;
    *arr = malloc(n * sizeof(int16_t));
    if (*arr != NULL) {
        res = 1;
        for (size_t i = 0; i < n; i++)
            scanf("%"SCNd16, &((*arr)[i]));
    }
    return res;
}

size_t input_n() {
    size_t n = 0;
    scanf("%zu", &n);
    return n;
}
