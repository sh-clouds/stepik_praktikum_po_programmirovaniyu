Вставка числа

Требуется вставить в данный массив на данное место данный элемент, сдвинув остальные элементы вправо.

Входные данные
В первой строке вводится одно натуральное число, не превосходящее 1000 – размер массива. Во второй строке задаются N чисел – элементы массива (целые числа, не превосходящие по модулю 1000). В третьей строке вводится число, которое необходимо вставить, и номер места, на которое его нужно вставить.

Выходные данные
Вывести получившийся массив.

Sample Input:
5
1 2 3 4 5
2 3

Sample Output:
1 2 2 3 4 5

Напишите программу. Тестируется через stdin → stdout

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int number = Integer.parseInt(input.nextLine());
        ++number;

        int[] array = new int[number];

        for (int a = 0; a < number - 1; a++) {
            array[a] = input.nextInt();
        }

        int value = input.nextInt();
        int position = input.nextInt();

        for (int b = number - 1; b > position - 1; b--) {
            array[b] = array[b - 1];
        }

        array[position - 1] = value;

        for (int b = 0; b < number; b++) {
            if(b < number - 1) System.out.print(array[b] + " ");
            else System.out.println(array[b]);
        }
    }
}

#include <inttypes.h>
#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

size_t input_n();
int8_t init_arr(int16_t** arr, size_t n);
void echo_arr(int16_t** arr, size_t n);
void add_in_arr(int16_t** arr, size_t* n);

int main() {
    size_t n = input_n();
    int16_t* array = NULL;
    if (n > 0 && init_arr(&array, n)) {
        add_in_arr(&array, &n);
        echo_arr(&array, n);
        free(array);
    }
    return 0;
}

void add_in_arr(int16_t** arr, size_t* n) {
    int16_t item;
    size_t position;
    scanf("%"SCNd16" %zu", &item, &position);
    *arr = realloc((*arr), (++(*n)) * sizeof(int16_t));
    for (size_t i = (*n); i > position; i--)
        (*arr)[i - 1] = (*arr)[i - 2];
    (*arr)[position - 1] = item;
}

void echo_arr(int16_t** arr, size_t n) {
    for (size_t i = 0; i < n; i++) {
        printf("%"PRId16, (*arr)[i]);
        if (i + 1 != n) printf(" ");
    }
}

int8_t init_arr(int16_t** arr, size_t n) {
    int8_t res = 0;
    *arr = malloc(n * sizeof(int16_t));
    if (*arr != NULL) {
        res = 1;
        for (size_t i = 0; i < n; i++)
            scanf("%"SCNd16, &((*arr)[i]));
    }
    return res;
}

size_t input_n() {
    size_t n = 0;
    scanf("%zu", &n);
    return n;
}
