Количество палиндромов

Назовем число палиндромом, если оно не меняется при перестановке его цифр в обратном порядке. Напишите программу, которая по заданному числу K выводит количество натуральных палиндромов, не превосходящих K.

Входные данные
Задано единственное число K (1 ≤ K ≤ 100000).

Выходные данные
Необходимо вывести количество натуральных палиндромов, не превосходящих K.

Sample Input:
1

Sample Output:
1

# java
import java.util.Scanner;
import java.util.stream.IntStream;

class Main {
    public static void main(String[] args) {
        try (Scanner scanner = new Scanner(System.in)) {
            int k = scanner.nextInt();
            System.out.println(IntStream.rangeClosed(1, k)
                                        .filter(Main::isPalindrome)
                                        .count());
        }
    }

    private static boolean isPalindrome(int n) {
        int copy = n;
        int reversed = 0;
        
        while (copy > 0) {
            reversed = reversed * 10 + copy % 10;
            copy /= 10;
        }

        return reversed == n;
    }
}

#include <iostream>

bool is_palindrom(int n) {
    int n2 = n, res = 0;
    std::cin >> n;
    while (n2) {
        res = res * 10 + n2 % 10;
        n2 /= 10;
    }
    return n == res;
}

int main() {
    int n, c = 0;
    std::cin >> n;
    for (int i = 1; i <= n; i++) {
        c += is_palindrom(i);
    }
    std::cout << c << "\n";
    return 0;
}
