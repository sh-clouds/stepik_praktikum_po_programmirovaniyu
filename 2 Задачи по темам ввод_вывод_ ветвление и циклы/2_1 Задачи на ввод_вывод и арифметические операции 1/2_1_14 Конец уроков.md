Конец уроков

В некоторой школе занятия начинаются в 9:00. Продолжительность урока — 45 минут, после 1-го, 3-го, 5-го и т.д. уроков перемена 5 минут, а после 2-го, 4-го, 6-го и т.д. — 15 минут. Определите, когда заканчивается указанный урок.

Входные данные
Дан номер урока (число от 1 до 10).

Выходные данные
Выведите два целых числа: время окончания урока в часах и минутах. При решении этой задачи нельзя пользоваться циклами и условными инструкциями.

Sample Input:
3

Sample Output:
11 35

# C
#include <stdio.h>

typedef struct {
    int hour;
    int minute;
} Time;

Time lesson_end_time(int lesson_number) {
    const int start_hour = 9;
    const int lesson_duration = 45;
    int breaks[] = {0, 5, 15, 5, 15, 5, 15, 5, 15, 5};
    
    int total_minutes = lesson_duration * lesson_number;
    
    for (int i = 0; i < lesson_number; i++) {
        total_minutes += breaks[i];
    }
    
    Time end_time;
    end_time.hour = start_hour + total_minutes / 60;
    end_time.minute = total_minutes % 60;
    
    return end_time;
}

int main(void) {
    int lesson_number;
    
    if (scanf("%d", &lesson_number) != 1) {
        return 1;
    }
    
    Time end_time = lesson_end_time(lesson_number);
    
    printf("%d %d", end_time.hour, end_time.minute);
    
 
    return 0;
}
