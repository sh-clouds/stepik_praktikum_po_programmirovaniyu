Апельсины бочками

Бизнесмен Василий после прочтения известной книги решил открыть новый бизнес – отгружать апельсины бочками. Партнерам важно знать сколько именно бочек апельсинов отгружается каждый день.

Мобильный телефон Василия поддерживает только транслит, поэтому он передает сообщения вида "N bochek" . Например, "3 bochki" или "1 bochka" .
Напишите программу, которая выбирает правильное слово (из "bochka" , "bochek" , "bochki" ) в зависимости от N.

Входные данные
Одно число N (0 ≤ N ≤ 1000).

Выходные данные
Фраза на транслите (см. примеры).

Sample Input:
15

Sample Output:
15 bochek

Напишите программу. Тестируется через stdin → stdout

# java
import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int a = in.nextInt();
        in.close();
        if (a % 100 >= 5 && a % 100 <= 20) {
            System.out.println(a + " bochek");
        } else if (a % 10 == 1) {
            System.out.println(a + " bochka");
        } else if (a % 10 >= 2 && a % 10 <= 4) {
            System.out.println(a + " bochki");
        } else {
            System.out.println(a + " bochek");
        }
    }
}


# C
#include <stdio.h>

int main() {
    int num; scanf("%d", &num);
    int last = num % 100;
    
    if (11 <= last && last <= 14) {
        printf("%d bochek", num);
        return 0;
    }
    switch(last % 10) {
        case 1:
            printf("%d bochka", num);
            break;
        case 2: case 3: case 4:
            printf("%d bochki", num);
            break;
        default:
            printf("%d bochek", num);
    }
    return 0;
}
