Число сочетаний

По данным натуральным n и k вычислите значение   Cnk=k!(n−k)!n! (число сочетаний из n элементов по k).

Входные данные
Вводятся 2 числа - n и k (n > k; n, k <= 30 ).

Выходные данные
Необходимо вывести  значение Cnk .

Sample Input:
2
1

Sample Output:
2

function getResult(n, k){
    return getFactorial(n)/(getFactorial(k) * getFactorial(n - k));
}

function getFactorial(n){
    if(n == 1) return 1
    else return getFactorial(n - 1) * n;
}

#include <iostream>
#include <string>
#include <vector>
using namespace std;

int sochetaniya(int n, int k) {
    long int result;
    vector <int> chislitel, znamenatel;
    int nk = n - k;

    if (k >= nk) {
        //кладём числа в вектор чтобы потом их перемножить
        for (int i = k + 1; i <= n; i++) {
            chislitel.push_back(i);
        }

        //кладём числа в знаменатель, тоже будем перемножать их
        for (int i = 2; i <= nk; i++) {
            znamenatel.push_back(i);
        }
    }
    else {//k < nk
        for (int i = nk + 1; i <= n; i++) {
            chislitel.push_back(i);
        }

        //кладём числа в знаменатель, тоже будем перемножать их
        for (int i = 2; i <= k; i++) {
            znamenatel.push_back(i);
        }
    }
       
       //сокращаем числа в числителе и знаменателе
       for (auto& x_chis : chislitel) {
           for (auto& y_znam : znamenatel) {
               if (x_chis % y_znam == 0 && y_znam != 1) {
                   x_chis /= y_znam;
                   y_znam = 1;
               }
               if (x_chis % 2 == 0 && y_znam % 2 == 0) {
                   x_chis /= 2;
                   y_znam /= 2;
               }
               if (x_chis % 5 == 0 && y_znam % 5 == 0) {
                   x_chis /= 5;
                   y_znam /= 5;
               }
               if (x_chis % 3 == 0 && y_znam % 3 == 0) {
                   x_chis /= 3;
                   y_znam /= 3;
               }
               if (x_chis % 7 == 0 && y_znam % 7 == 0) {
                   x_chis /= 7;
                   y_znam /= 7;
               }
           }
       }

       //сокращаем числа в числителе и знаменателе
       for (auto& x_chis : chislitel) {
           for (auto& y_znam : znamenatel) {
               if (x_chis % y_znam == 0 && y_znam != 1) {
                   x_chis /= y_znam;
                   y_znam = 1;
               }
           }
       }

       long int chislo_v_chislitele = 1;
       for (auto chislo_v_vectore : chislitel) {
           chislo_v_chislitele *= chislo_v_vectore;
       }

       long int chislo_v_znamenatele = 1;
       for (auto chislo_v_vectore : znamenatel) {
           chislo_v_znamenatele *= chislo_v_vectore;
       }

       result = chislo_v_chislitele / chislo_v_znamenatele;
    return result;
}
