# java
import java.util.*;

class Main {
    public static void main(String[] args) {
        System.out.print(Arrays.stream(new Scanner(System.in).nextLine().split(" "))
                .mapToInt(Integer::parseInt)
                .filter(e -> e > 0)
                .count());
    }
}

        N++;
        if (N == 1) array = malloc(sizeof(int64_t));
        else array = realloc(array, N * sizeof(int64_t));
        array[N - 1] = num;
    }
    
    size_t count = 0;
    for (size_t i = 0; i < N; i++)
        if (array[i] > 0) count++;
    printf("%zu", count);
    
    if (N) free(array);
    return 0;
}


# C++
#include <iostream>
#include <vector>
#include <cmath>

int main() {
    // put your code here
    std::vector <int> arr;
    
    int temp;
    while(std::cin >> temp)
    {
        arr.push_back(temp);
    }   
        
    int count = 0;
    
    //Считаем отрицательные числа
    //std::signbit возвращает true для -1
    //и false для 1
    for(auto &ptr : arr)
    {
        count += std::signbit(ptr);
        if(ptr == 0)
            ++count;
    }
    
    //Выводим кол-во положительных
    std::cout << arr.size() - count << std::endl;
    return 0;
}
