Таблица умножения

Даны два числа n и m. Создайте двумерный массив A[n][m], заполните его таблицей умножения A[i][j]=i*j и выведите на экран. При этом нельзя использовать вложенные циклы, все заполнение массива должно производиться одним циклом.

Входные данные
Программа получает на вход два числа n и m – количество строк и столбцов, соответственно.

Выходные данные
Программа должна вывести  полученный массив. Числа разделяйте одним пробелом.

Sample Input:
3 3

Sample Output:
0 0 0
0 1 2
0 2 4

# ruby
n,m = gets.chomp.split(' ').map { |j| j.to_i }
num_row, num_col = 0, 0
a = []

0.upto(n-1) do |i|
    a.push([])
end

0.upto(n*m-1) do |i|
    num_row = i / m
    num_col = i - (num_row * m)
    a[num_row].push(num_row * num_col)
end

a.each do |i|
    puts i.join(" ")
end 


# python
n, m = map(int, input().split())
matrix = [[0] * m for _ in range(n)]
for k in range(n * m):
    i = k // m
    j = k % m
    matrix[i][j] = i * j
for row in matrix:
    print(*row)

using System;

public class MainClass
{
    public static void Main()
    {
        int[] nm = Array.ConvertAll(Console.ReadLine().Split(), int.Parse);
        int n = nm[0];
        int m = nm[1];
        int[][] arr = new int[n][];

        for (int y = 0; y < n; y++)
        {
            arr[y]=(GetRange(m,y));
        }
        for(int y=0;y<n;y++)
        {
            for( int x=0;x<m;x++)
            {
                Console.Write(x<m-1?arr[y][x]+" ":arr[y][x]);
            }
            Console.WriteLine();
        }        
    }
    public static int[] GetRange(int items, int step)
    {
        int[] arr = new int[items];
        for (int i = 0; i < items; i++)
        {
            arr[i] = i * step;
        }
        return arr;
    }
}
