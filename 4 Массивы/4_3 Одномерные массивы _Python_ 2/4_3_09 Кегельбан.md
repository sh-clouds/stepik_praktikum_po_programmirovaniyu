Кегельбан

N кеглей выставили в один ряд, занумеровав их слева направо числами от 1 до N. Затем по этому ряду бросили K шаров, при этом i-й шар сбил все кегли с номерами от lili до riri включительно. Определите, какие кегли остались стоять на месте.
Входные данные
Программа получает на вход количество кеглей N и количество бросков K. Далее идет K пар чисел lili, riri, при этом 1≤lili≤riri≤N.
Выходные данные
Программа должна вывести последовательность из N символов, где j-й символ есть “I”, если j-я кегля осталась стоять, или “.”, если j-я кегля была сбита.
Sample Input:
10 3
8 10
2 5
3 6
Sample Output:
I.....I...

# python
n, k = map(int, input().split())
res = ["I"] * n
while (k := k - 1) >= 0:
    a, b = map(int, input().split())
    for i in range(a - 1, b):
        res[i] = "."
print(*res, sep='')

void fill_array(char** arr, size_t N);
void shot(char** arr, size_t N);
void output(char** arr, size_t N);

int main() {
    size_t N = 0, shots = 0;
    char* array;
    input(&N, &shots);
    N++;
    fill_array(&array, N);
    for (size_t i = 0; i < shots; i++) shot(&array, N);
    output(&array, N);
    free(array);
    return 0;
}

void output(char** arr, size_t N) {
    for (size_t i = 1; i < N; i++) printf("%c", (*arr)[i]);
}

void shot(char** arr, size_t N) {
    size_t A, B;
    scanf("%zu", &A);
    scanf("%zu", &B);
    for (size_t i = A; i <= B; i++) (*arr)[i] = '.';
}

void fill_array(char** arr, size_t N) {
    (*arr) = malloc((N + 1) * sizeof(int64_t));
    (*arr)[0] = '-';
    for (size_t i = 1; i < N; i++) (*arr)[i] = 'I';
}

void input(size_t* N, size_t* X) {
    scanf("%zu", &(*N));
    scanf("%zu", &(*X));
}
