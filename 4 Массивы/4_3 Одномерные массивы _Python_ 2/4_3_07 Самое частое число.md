Самое частое число

Дан список. Не изменяя его и не используя дополнительные списки, определите, какое число в этом списке встречается чаще всего.

Если таких чисел несколько, выведите любое из них.

Входные данные
Вводится список чисел. Все числа списка находятся на одной строке.

Выходные данные
Выведите ответ на задачу.

Sample Input:
1 2 3 2 3 3

Sample Output:
3

Напишите программу. Тестируется через stdin → stdout

      a[c] = (a[c] || 0) + 1;
      return a;
    }, {});
    let maxCount = Math.max(...Object.values(counts));

size_t count_item(int64_t** arr, size_t N, int64_t X) {
    size_t count = 0;
    for (size_t i = 0; i < N; i++)
        if ((*arr)[i] == X) count++;
    return count;
}

void append_item(int64_t** arr, size_t* N, size_t K, int64_t X) {
    (*N)++;
    (*arr) = realloc((*arr), (*N) * sizeof(int64_t));
    for (size_t i = (*N) - 1; i > K; i--) (*arr)[i] = (*arr)[i - 1];
    (*arr)[K] = X;
}

void rem_item(int64_t** arr, size_t* N, size_t K) {
    for (size_t i = K + 1; i < (*N); i++) (*arr)[i - 1] = (*arr)[i];
    pop_item(&(*arr), &(*N));
}

void pop_item(int64_t** arr, size_t* N) {
    (*N)--;
    (*arr) = realloc((*arr), (*N) * sizeof(int64_t));
}

void output(int64_t** arr, size_t N) {
    for (size_t i = 0; i < N; i++) {
        printf("%"PRId64, (*arr)[i]);
        if (i != N - 1) printf(" ");
    }
}

void input(int64_t** arr, size_t* N, size_t* K, int64_t* X) {

        (*arr)[(*N) - 1] = num;
    }
    (*X) = 0;
    (*K) = 0;
}
